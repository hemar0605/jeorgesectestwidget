const readline = require('readline'); const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); function calculator() { rl.question('Enter an expression: ', (input) => { try { const result = eval(input); console.log(`Result: ${result}`); } catch (error) { console.error(`Error: ${error.message}`); } calculator(); }); } console.log('Welcome to the calculator!'); calculator();
